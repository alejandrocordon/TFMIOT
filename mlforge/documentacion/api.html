Referencia de los modulos y clases principales de MLForge para uso programatico.

mlforge.config

Carga y validacion de configuraciones YAML.

load_config(path) -> ProjectConfig

Carga un archivo YAML y devuelve un objeto de configuracion tipado.

from mlforge.config import load_config

config = load_config("config.yaml")
print(config.model.architecture)    # "mobilenet_v3_small"
print(config.training.epochs)        # 20
print(config.data.dataset)           # "flowers102"

Dataclasses de configuracion

ClaseCampos principales
ProjectConfigname, task, data, model, training, export
DataConfigsource, dataset, input_size, augmentation, split, data_dir
ModelConfigarchitecture, pretrained, num_classes, framework
TrainingConfigepochs, batch_size, optimizer, learning_rate, scheduler, early_stopping, output_dir, device
ExportConfigformats, output_dir

    # Tu modelo aqui

detector.py

from mlforge.models.detector import (

mlforge.training

Motores de entrenamiento para PyTorch y TensorFlow.

trainer_pytorch.train(config) -> Path

from mlforge.config import load_config
from mlforge.training.trainer_pytorch import train

config = load_config("config.yaml")
output_dir = train(config)
# output_dir contiene: best_model.pth, final_model.pth, metrics.json

trainer_tf.train(config) -> Path

from mlforge.training.trainer_tf import train

# Misma interfaz, seleccion automatica por config.model.framework
output_dir = train(config)
# output_dir contiene: best_model.keras, final_model.keras, metrics.json

callbacks.py

from mlforge.training.callbacks import EarlyStopping, ModelCheckpoint, Callback

# EarlyStopping: para training si no mejora en N epochs
es = EarlyStopping(patience=5, metric="val_accuracy", higher_is_better=True)

# ModelCheckpoint: guarda el mejor modelo automaticamente
ckpt = ModelCheckpoint(output_dir="./outputs", metric="val_accuracy")

# Custom callback: heredar de Callback
class MiCallback(Callback):
    def on_epoch_end(self, model, metrics):
        print(f"Epoch {metrics.epoch}: {metrics.val_accuracy:.2%}")
        return True  # False para parar

metrics.py

from mlforge.training.metrics import EpochMetrics, MetricsTracker

# EpochMetrics: datos de una epoch
metrics = EpochMetrics(
    epoch=1, train_loss=0.5, train_accuracy=0.8,
    val_loss=0.3, val_accuracy=0.85,
    learning_rate=0.001, duration_seconds=12.3,
)

# MetricsTracker: persiste a JSON
tracker = MetricsTracker("./outputs")
tracker.log_epoch(metrics)
print(tracker.best_metric("val_accuracy"))  # 0.85

mlforge.export

Exportacion a multiples formatos con validacion y benchmark.

from mlforge.export.onnx_export import ONNXExporter
from mlforge.export.tflite_export import TFLiteExporter, EdgeTPUExporter
from mlforge.export.coreml_export import CoreMLExporter
from mlforge.export.tfjs_export import TFJSExporter
from mlforge.export.benchmark import run_benchmark
from mlforge.export.quantize import quantize_dynamic

# Exportar a ONNX
exporter = ONNXExporter(model, input_size=224)
onnx_path = exporter.export("./exported")
exporter.validate(onnx_path)  # Compara outputs PyTorch vs ONNX

# Benchmark comparativo
run_benchmark("./exported", input_size=224, n_runs=100)

mlforge.deploy

Scaffolding de proyectos de despliegue y servidor de inferencia.

scaffold.py

from mlforge.deploy.scaffold import scaffold, list_targets

# Ver targets disponibles
for t in list_targets():
    print(f"{t['name']}: {t['description']}")

# Generar proyecto Android
project = scaffold(
    target="android",
    output_dir="./deploy",
    model_path="model.tflite",
    labels=["rosa", "tulipan", "girasol"],
    input_size=224,
)

server.py

from mlforge.deploy.server import create_app, ONNXPredictor

# Crear predictor standalone
predictor = ONNXPredictor(
    "model.onnx",
    labels=["rosa", "tulipan"],
    input_size=224,
)

# Predecir
from PIL import Image
img = Image.open("flower.jpg")
results = predictor.predict(img, top_k=5)
# [{"label": "rosa", "confidence": 0.92, "class_id": 0}, ...]

# Crear FastAPI app
app = create_app(model_path="model.onnx", labels=["rosa", "tulipan"])
# Servir con uvicorn
